*** Generate getter and setters for MongoDocuments ***
./mongodb odm:generate:documents app/models

*** Create or update the database from annotations (Note: this does not change existing documents) ***
./mongodb odm:schema:create

*** Clear the doctrine cache ... this is now erroring with "No Metadata cache driver is configured on given DocumentManager." for some reason. ***
./mongodb odm:clear-cache:metadata

*** Generate API docs ***
php vendor/jimgitsit/documen/documen.php generate-docs

*** Copy the remote dev db to local ***
./dbcopy-dev-local.sh

*** SSH tunnel port 27018 to dev mongodb ***
ssh -L 27018:localhost:27017 jim@dev.alacarte.bluetree.ws

*** backup alacarte-prod mongo db ***
(from project root, unless you want to put the backup somewhere else)
mongodump -d alacarte-prod -o db/alacarte-prod

*** Restore alacarte-prod mongo db ***
(from project root, unless you want to put the backup somewhere else)
mongorestore -d alacarte-prod --drop /db/alacarte-prod

*** Build the add-on .xpi file (Enable unsigned add-ons in Firefox: http://winaero.com/blog/enable-unsigned-add-ons-disabled-by-firefox-43/) ***
jpm xpi

jpm post --post-url http://localhost:8888/
